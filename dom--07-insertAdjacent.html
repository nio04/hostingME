New Way

model

the full model is shown below

              
                  beforebegin

                  #target element tag #start

                    afterbegin

                    ##target element content

                    beforeend

                  #target element tag #end

                  afterend

beforebegin

insert as sibling element, just before the target element tag. 

              
                const target = document.querySelector("#target");
                const element = document.querySelector("#element");

                target.insertAdjacentElement("beforebegin", element);

afterbegin

insert as child element, just before the target element content 

						
							const target = document.querySelector("#target"); const
							element = document.querySelector("#element");
							target.insertAdjacentElement("afterbegin", element);

beforeend

insert as child element, just after the target element content. 

						
							const target = document.querySelector("#target"); const
							element = document.querySelector("#element");
							target.insertAdjacentElement("beforeend", element);

afterend

insert as sibling element, just after the target element tag. 

               
                const target = document.querySelector("#target");
                const element = document.querySelector("#element");

                target.insertAdjacentElement("afterend", element);
