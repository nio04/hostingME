center a div by css

flexbox

              
                .parent {
                  display: flex;
                  justify-content: center;
                  align-items: center;
                  height: 100vh;
                }

grid

              
                .parent {
                  display: grid;
                  place-items: center;
                  height: 100vh;
                }

absolute Positioning

              
                .parent {
                position: relative;
                height: 100vh;
              }

              .child {
                position: absolute;
                top: 50%;
                left: 50%;
                transform: translate(-50%, -50%);
              }

css calc() [known width && height]

Center horizontally with known width 

              
               .child {
                position: absolute;
                left: 50%;
                width: 200px;
                margin-left: -100px;
              }

Center vertically with known height 

              
               .child {
                position: absolute;
                top: 50%;
                height: 100px;
                margin-top: -50px;
              }

grid && auto-margin

              
               .container {
                display: grid;
              }

              .centered {
                margin: auto;
              }

text-align && line-height

              
               .container {
                text-align: center;
                height: 100vh;
              }

              .centered {
                display: inline-block;
                vertical-align: middle;
                line-height: 100vh;
              }
